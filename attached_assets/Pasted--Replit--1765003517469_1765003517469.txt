STEP 1 – CREATE A REUSABLE COPY HOOK
========================================
Create a small hook so we can reuse the same pattern in multiple places.

1. Add a new file (adjust path if there is a hooks folder):

   `client/src/hooks/useCopyFeedback.ts`

2. Implement:

```ts
import { useState, useCallback } from "react";

export function useCopyFeedback(timeoutMs: number = 2000) {
  const [isCopied, setIsCopied] = useState(false);

  const copyWithFeedback = useCallback(async (text: string) => {
    try {
      await navigator.clipboard.writeText(text);
      setIsCopied(true);
      window.setTimeout(() => {
        setIsCopied(false);
      }, timeoutMs);
    } catch (err) {
      console.error("Failed to copy", err);
      // Optional: in the future we can add error feedback.
    }
  }, [timeoutMs]);

  return { isCopied, copyWithFeedback };
}

	•	Do not remove existing helpers yet; we will integrate them with this hook.
