import React, { useEffect, useState } from "react";
import { SplashScreen } from "./components/SplashScreen"; // adjust relative path
// ... keep other existing imports (logos, cards, themes, etc.)

const HomePage: React.FC = () => {
  const [showSplash, setShowSplash] = useState(true);

  useEffect(() => {
    const timer = setTimeout(() => {
      setShowSplash(false);
    }, 3000); // ~3 seconds

    return () => clearTimeout(timer);
  }, []);

    <>

           - Theme toggle (Arcade 90s / Premium)

      </main>
    </>
  );
};
